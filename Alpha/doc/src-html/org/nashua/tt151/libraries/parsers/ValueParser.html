004/**
005 * A class containing methods for converting doubles and ints to adhere to the
006 * protocol. The class can convert doubles and integers to message-format and 
007 * contains wrappers for safely parsing strings
008 * @author Brennan
009 * @version 1.1
010 */
011public final class ValueParser {
012    
013    //Prevent Instantiation
014    private ValueParser() { }
015    
016    /**
017     * Clamps an integer and converts it to a string
018     * @param num Integer to convert
019     * @param min Minimum the number can be
020     * @param max Maximum the number can be
021     * @return A string representing the number
022     */
023    public static String Convert(int num, int min, int max) {
024        //Clamp
025        if (num > max) { num = max; }
026        if (num < min) { num = min; }
027        return Integer.toString(num);
028    }
029    
030    /**
031     * Clamps a double, truncates it and converts it to a string
032     * @param num Double to convert
033     * @param min Minimum to clamp
034     * @param max Maximum to clamp
035     * @param places Number of decimal places (rounded)
036     * @return 
037     */
038    public static String Convert(double num, double min, double max, int places) {
039        //Clamping
040        if (num > max) { num = max; }
041        if (num < min) { num = min; }
042        //Truncating
043        //BigDecimal bd = new BigDecimal(num);
044        //bd = bd.setScale(places, BigDecimal.ROUND_HALF_UP);
045        //return bd.doubleValue() + "";
046        return "" + num;
047    }
048    
049    /**
050     * Safely parses a string to an integer.
051     * @param val String representing an integer
052     * @return An integer representing the string, -1 is returned if the string cannot be parsed
053     */
054    public static int ParseInt(String val) {
055        int num;
056        
057        try {
058            num = Integer.parseInt(val);
059        } catch (NumberFormatException ex) {
060            System.out.println(ex.toString());
061            num = -1;
062        }
063        
064        return num;
065    }
066    
067    /**
068     * Safely parses a string to a double.
069     * @param val String to parse
070     * @return A double best represented by the string, Double.NaN is returned upon error
071     */
072    public static double ParseDouble(String val) {
073        double d;
074        
075        try {
076            d = Double.parseDouble(val);
077        } catch (NumberFormatException ex) {
078            System.out.println(ex.toString());
079            d = Double.NaN;
080        }
081        
082        return d;
083    }
084
085
086}

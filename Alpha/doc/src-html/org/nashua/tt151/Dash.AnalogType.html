032    /**
033     * Query the dashboard for the speed (in ft/s) to set the shooter to
034     * @param t Target ti get speed for
035     * @return The speed (in ft/s) to set the shooter to or -1 on failure
036     * @throws NumberFormatException
037     * @throws IOException 
038     */
039    public double queryShooterSpeed(Target t) throws NumberFormatException, IOException {
040        /* Send query to the dashboard
041         * No Value
042         * Arguments:
043         *   Analog Device Type
044         */
045        con.send(ProtocolParsing.StringParser.createMessage(ProtocolParsing.Command.Query, 
046                ProtocolParsing.Key.ShooterSpeed, "", new String[]{""+t.getID()}));
047        // Wait for result
048        String[][] result = null;
049        while ((result=con.getKeyFromQueue(ProtocolParsing.Key.ShooterSpeed))==null) {
050        }
051        // Verify non null value and parse the double from the string
052        if (result.length>1 && result[1].length>0) {
053            return Double.parseDouble(result[1][0]);
054        }
055        // Return -1 on null result
056        return -1;
057    }

090    /**
091     * Send the value of the encoder to the dashboard for displaying
092     * @param distance Distance the encoder has traveled (in inches)
093     * @param linearSpeed Linear speed of the encoder (in ft/s)
094     * @param wheelDiameter Diameter of the wheel (used to calculate RPM)
095     * @param name Name of the encoder to display on the dashboard
096     * @param mod1 Module of the first PWM (1 or 2)
097     * @param slot1 Slot of the first PWM (1 through 14)
098     * @param mod2 Module of the second PWM (1 or 2)
099     * @param slot2 Slot of the second PWM (1 through 14)
100     * @throws IOException 
101     */
102    public void sendEncoder(double distance, double linearSpeed, double wheelDiameter, String name, int mod1, 
103            int slot1, int mod2, int slot2) throws IOException {
104        /* Send the encoder value to the dashboard
105         * Value: distance (in inches) - 2 Decimal Places
106         * Arguments:
107         *   Linear Speed (in ft/s) - 2 Decimal Places
108         *   Wheel Diameter (in inches) - 2 Decimal Places
109         *   Name
110         *   Module1 Slot1 (###)
111         *   Module2 Slot2 (###)
112         */
113        distance = MathTools.round(distance, 2);
114        linearSpeed = MathTools.round(linearSpeed, 2);
115        wheelDiameter = MathTools.round(wheelDiameter, 2);
116        con.send(ProtocolParsing.StringParser.createMessage(ProtocolParsing.Command.Send, 
117                ProtocolParsing.Key.EncoderValue, ""+distance, new String[]{""+linearSpeed, ""+wheelDiameter, name, 
118                    ""+mod1+(slot1<10?"0"+slot1:""+slot1), ""+mod2+(slot2<10?"0"+slot2:""+slot2)}));
119    }

282    /**
283     * The ConnectionListener interface has the method declarations for the 
284     * methods that get called when an event happens on the communication
285     * channel. The following events are currently supported: onConnect,
286     * onDisconnect, and onDataReceived.
287     * @author Brian Ashworth
288     * @version 1.0
289     */
290    public static interface ConnectionListener {
291        /**
292         * This listener method gets called when a connection is made
293         */
294        public void onConnect();
295        /**
296         * This listener method gets called when a connection is lost
297         */
298        public void onDisconnect();
299        /**
300         * This listener method gets called when a message from the dashboard is received.
301         * @param msg Message from the dashboard
302         */
303        public void onDataReceived(String msg);
304    }
305    /**
306     * The Connection class establishes the communication channel from the robot
307     * to the laptop (and vise versa).
308     * @author Brian Ashworth
309     * @version 1.0
310     */
311    private static class Connection implements ConnectionListener {
312        /**
313         * Listener for this connection
314         */
315        private ConnectionListener listener;
316        /**
317         * Socket connection between the laptop and the cRIO. This is the client side.
318         */
319        private SocketConnection socket;
320        /**
321         * Reader for the input stream. The input stream is to the cRIO
322         */
323        private InputStreamReader reader;
324        /**
325         * The Writer for the output stream. The output stream is from the cRIO.
326         */
327        private BufferedWriter writer;
328        /**
329         * Queue of queried items
330         */
331        private String[][][] queue = new String[0][0][0];
332        /**
333         * Establish a client connection from the cRIO to the dashboard
334         * @param host IP Address of laptop
335         * @param port Port that the dashboard is listening on
336         * @param cl Listener for this connection
337         * @throws IOException 
338         */
339        public Connection(String host, int port, ConnectionListener cl) throws IOException {
340            // Set listener if null to this (dummy) or if not null to the listener
341            listener = cl==null?this:cl;
342            
343            // Connect to server (in this case the dashboard on the laptop)
344            socket = (SocketConnection) Connector.open("socket://"+host+":"+port);
345            reader = new InputStreamReader(socket.openDataInputStream());
346            writer = new BufferedWriter(new OutputStreamWriter(socket.openDataOutputStream()));
347            
348            // Listen for events every 10ms
349            final java.util.Timer t = new java.util.Timer();
350            t.schedule(new java.util.TimerTask(){
351                public void run(){
352                    if (socket==null) {
353                        t.cancel();
354                        return;
355                    }
356                    try {
357                        if (reader.ready()) {
358                            // Read length byte
359                            int length = reader.read();
360                            // Read for length bytes
361                            int offset = 0;
362                            char[] buffer = new char[length];
363                            while (length!=0) {
364                                int read = reader.read(buffer, offset, length);
365                                offset+=read;
366                                length-=read;
367                                if (read==-1) {
368                                    break;
369                                }
370                            }
371                            // Turn buffer into String
372                            String msg = String.valueOf(buffer);
373                            // Check to see if disconnected
374                            if (msg.equals("[Disconnected]")) {
375                               socket = null;
376                               reader = null;
377                               writer = null;
378                               listener.onDisconnect();
379                            } else if (ProtocolParsing.StringParser.isReply(msg)) {
380                                // Add reply to the queue
381                                String key = ProtocolParsing.StringParser.getKey(msg).shorthand;
382                                String value = ProtocolParsing.StringParser.getValue(msg);
383                                String[] args = ProtocolParsing.StringParser.getArgs(msg);
384                                updateQueue(key, value, args);
385                            } else {
386                                // Send to the connection listener
387                                listener.onDataReceived(msg);
388                            }
389                        }
390                    } catch (Exception e) {
391                        System.err.println(e.getClass()+": "+e.getMessage());
392                    }
393                }
394            },1,10);
395            
396            listener.onConnect();
397        }
398
399        /**
400         * Remove all null indices from queue and shrink down to size
401         */
402        private void cleanQueue() {
403            // Remove null indices from queue
404            int nc = 0;
405            for (int i=0;i<queue.length;i++) {
406                if (queue[i]==null) {
407                    nc++;
408                }
409            }
410            String[][][] newQueue = new String[queue.length-nc][3][0];
411            int index = 0;
412            for (int i=0;i<queue.length;i++) {
413                if (queue[i]!=null) {
414                    newQueue[index++] = queue[i];
415                }
416            }
417        }
418        /**
419         * Retrieve a key from the queue
420         * @param key Key to retrieve
421         * @return The value (and args) that correspond to the key or null if not found
422         */
423        public String[][] getKeyFromQueue(ProtocolParsing.Key key) {
424            return getKeyFromQueue(key.shorthand);
425        }
426        /**
427         * Retrieve a key from the queue
428         * @param key Key to retrieve
429         * @return The value (and args) that correspond to the key or null if not found
430         */
431        public String[][] getKeyFromQueue(String key) {
432            for (int i=0; i<queue.length; i++) {
433                if (queue[i]!=null&&queue[i][0]!=null&&queue[i][0][0].equals(key)) {
434                    String[][] value = queue[i];
435                    queue[i] = null;
436                    cleanQueue();
437                    return value;
438                }
439            }
440            return null;
441        }
442        /**
443         * Dummy onConnect listener method
444         */
445        public void onConnect() {
446        }
447
448        /**
449         * Dummy onDisconnect listener method
450         */
451        public void onDisconnect() {
452        }
453
454        /**
455         * Dummy onDataReceived listener method
456         * @param msg Message received
457         */
458        public void onDataReceived(String msg) {
459        }
460        
461        /**
462         * Send message to the dashboard
463         * @param msg Message to send
464         * @throws IOException 
465         */
466        public void send(String msg) throws IOException {
467            writer.write(""+(char)msg.length());
468            writer.write(msg);
469            writer.flush();
470        }
471        
472        /**
473         * Add item to the queue
474         * @param key Key of the queued item
475         * @param val Value of the queued item
476         * @param args Args of the queued item
477         */
478        private void updateQueue(String key, String val, String[] args) {
479            String[][][] newQueue = new String[queue.length+1][3][0];
480            System.arraycopy(queue, 0, newQueue, 0, queue.length);
481            newQueue[queue.length] = new String[][]{new String[]{key}, new String[]{val}, args};
482            queue = newQueue;
483            cleanQueue();
484        }
485    }

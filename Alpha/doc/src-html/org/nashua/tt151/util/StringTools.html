011public class StringTools {
012    
013    private StringTools() {}
014    
015    //Helper method used for counting the occurances of a character in a String
016    public static int count(String msg, char sep) {
017        int c = 0;
018        for (int i=0;i<msg.length();i++) {
019            if (msg.charAt(i)==sep) {
020                c++;
021            }
022        }
023        return c;
024    }
025
026    //Helper method used for splitting a String based on a character
027    //TODO Make it prettier
028    public static String[] split(String msg, char sep) {
029        String[] parts = new String[count(msg, sep)+1];
030        int index = -1;
031        parts[0] = msg.substring(0, msg.indexOf(sep, index+1));
032        for (int i=1;i<parts.length;i++) {
033            int findex = msg.indexOf(sep, index+1);
034            if (findex==-1) {
035                parts[i] = msg.substring(index+1);
036                break;
037            } else {
038                index=msg.indexOf(sep, findex+1);
039                if (index==-1) {
040                    parts[i] = msg.substring(findex+1);
041                } else {
042                    parts[i] = msg.substring(findex+1, index);
043                }
044            }
045        }
046        return parts;
047    }

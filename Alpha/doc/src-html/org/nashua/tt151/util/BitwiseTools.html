005package org.nashua.tt151.util;
006
007/**
008 *
009 * @author Brennan
010 */
011public class BitwiseTools {
012    
013    public static byte removebit(byte num, byte bit) {
014        byte power = (byte)MathTools.pow(2.0, bit);
015        if ((num & power) == power) {
016            num -= power;
017        }
018        
019        return num;
020    }
021    
022    public static byte removebits(byte num, byte bitstart, byte bitend) {
023       
024        for (int i = bitstart; i != bitend + 1; i++) {
025            num = removebit(num, (byte)i);
026        }
027        return num;
028    }
029    
030    /**
031     * Returns an int representing an unsigned byte. -128 is 0, -127 is 1, etc
032     * @param num A byte to unsign
033     * @return An int representing the byte unsigned;
034     */
035    public static int unsign(byte num) {
036        int s;
037        s = num + Byte.MAX_VALUE;
038        
039        return s;
040    }
041}

001package org.nashua.tt151;
002
003/**
004 * This class stores all the information on the various targets.
005 * @author Brian Ashworth
006 * @version 1.0
007 */
008public final class Target {
009    /**
010     * This is a fake target that allows for identifying the lowest visible target.
011     */
012    public static final Target LOWEST = new Target(0, 0, 0, 0);
013    /**
014     * This is the bottom target. The opening is 29"x24" and it is 19" off the ground.
015     */
016    public static final Target BOTTOM = new Target(1, 29, 24, 19);
017    /**
018     * This is the middle right target. The opening is 54"x21" and it is 88 5/8" off the ground.
019     */
020    public static final Target MIDDLE_RIGHT = new Target(2, 54, 21, 88.625);
021    /**
022     * This is the middle left target. The opening is 54"x21" and it is 88 5/8" off the ground.
023     */
024    public static final Target MIDDLE_LEFT = new Target(3, 54, 21, 88.625);
025    /**
026     * This is the top target. The opening is 54"x12" and it is 104 1/8" off the ground.
027     */
028    public static final Target TOP = new Target(4, 54, 12, 104.125);
029    /**
030     * This is a fake target that allows for identifying the highest visible target.
031     */
032    public static final Target HIGHEST = new Target(5, 0, 0, 0);
033    
034    private int id;
035    private double slotWidth;
036    private double slotHeight;
037    private double heightToSlot;
038    /**
039     * The Target class is used for identifying the targets on the field and their properties.
040     * @param id - Target ID
041     * @param sw - Width of slot in inches
042     * @param sh - Height of slot in inches
043     * @param hts - Height to slot in inches
044     */
045    private Target(int id, double sw, double sh, double hts) {
046        this.id = id;
047        this.slotWidth = sw;
048        this.slotHeight = sh;
049        this.heightToSlot = hts;
050    }
051    /**
052     * Retrieves the height to the center of the target
053     * @return The height to the center of the target in inches
054     */
055    public double getHeightToCenter() {
056        return getHeightToSlot()+getSlotHeight()/2.0;
057    }
058    /**
059     * Retrieves the height to the slot
060     * @return The height to the slot in inches
061     */
062    public double getHeightToSlot() {
063        return heightToSlot;
064    }
065    /**
066     * Retrieves the ID of the target. This is only used for communication
067     * between the cRIO and the dashboard on the laptop.
068     * @return Target ID
069     */
070    public int getID() {
071        return id;
072    }
073    /**
074     * Retrieves the height of the slot
075     * @return The height of the slot in inches
076     */
077    public double getSlotHeight() {
078        return slotHeight;
079    }
080    /**
081     * Retrieves the width of the slot
082     * @return The width of the slot in inches
083     */
084    public double getSlotWidth() {
085        return slotWidth;
086    }
087    /**
088     * Override of java.lang.Object.toString. A textual representation of the Target object.
089     * @return A textual representation of the Target object
090     */
091    public String toString() {
092        return "org.nashua.tt151.Target[id="+getID()+", slotWidth="+getSlotWidth()+", slotHeight="+getSlotHeight()
093                +", heightToSlot="+getHeightToSlot()+", heightToCenter="+getHeightToCenter()+"]";
094    }
095}
